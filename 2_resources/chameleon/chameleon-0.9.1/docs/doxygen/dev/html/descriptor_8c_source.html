258  **/
259 int morse_desc_check(MORSE_desc_t *desc)
260 {
261  if (desc == NULL) {
262  morse_error("morse_desc_check", "NULL descriptor");
263  return MORSE_ERR_NOT_INITIALIZED;
264  }
265  if (desc->mat == NULL && desc->use_mat == 1) {
266  morse_error("morse_desc_check", "NULL matrix pointer");
267  return MORSE_ERR_UNALLOCATED;
268  }
269  if (desc->dtyp != MorseRealFloat &&
270  desc->dtyp != MorseRealDouble &&
271  desc->dtyp != MorseComplexFloat &&
272  desc->dtyp != MorseComplexDouble ) {
273  morse_error("morse_desc_check", "invalid matrix type");
274  return MORSE_ERR_ILLEGAL_VALUE;
275  }
276  if (desc->mb <= 0 || desc->nb <= 0) {
277  morse_error("morse_desc_check", "negative tile dimension");
278  return MORSE_ERR_ILLEGAL_VALUE;
279  }
280  if (desc->bsiz < desc->mb*desc->nb) {
281  morse_error("morse_desc_check", "tile memory size smaller than the product of dimensions");
282  return MORSE_ERR_ILLEGAL_VALUE;
283  }
284  if (desc->lm <= 0 || desc->ln <= 0) {
285  morse_error("morse_desc_check", "negative matrix dimension");
286  return MORSE_ERR_ILLEGAL_VALUE;
287  }
288  if ((desc->lm < desc->m) || (desc->ln < desc->n)) {
289  morse_error("morse_desc_check", "matrix dimensions larger than leading dimensions");
290  return MORSE_ERR_ILLEGAL_VALUE;
291  }
292  if ((desc->i > 0 && desc->i >= desc->lm) || (desc->j > 0 && desc->j >= desc->ln)) {
293  morse_error("morse_desc_check", "beginning of the matrix out of scope");
294  return MORSE_ERR_ILLEGAL_VALUE;
295  }
296  if (desc->i+desc->m > desc->lm || desc->j+desc->n > desc->ln) {
297  morse_error("morse_desc_check", "submatrix out of scope");
298  return MORSE_ERR_ILLEGAL_VALUE;
299  }
300  return MORSE_SUCCESS;
301 }
